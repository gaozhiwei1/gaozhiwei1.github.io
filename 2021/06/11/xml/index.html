进而我们采用cv2包和xml.etree.ElmentTree包对其进行数据提取并且展示在图片上面如下

# -*- coding: utf-8 -*-
import xml.etree.ElementTree as ET
from xml.etree.ElementTree import Element, SubElement, ElementTree
import os
import cv2
import glob

class Xml:
    '''
    该类用于提取旧的xml中数据，并生成新的xml类

    Attributes:
         path:旧的xml路径
         vol: 旧的xml中不能直接看出的电压，需要人工识别
    '''
    def __init__(self,path,vol,save_path,img_path) -> None:
        self.path=path
        self.vol=vol
        self.save_path=save_path
        self.text=[] # 名称为text的列表
        self.label=[] #名称非text的列表
        self.img_path=img_path
    def old_xml_parse(self):
        '''
        该函数用于解析提取旧的xml中数据
        '''
        text_index=1
        label_index=1
        im=cv2.imread(self.img_path)
        tree=ET.parse(self.path)
        root=tree.getroot()
        color = (4, 250, 7)
        for object in root.findall('object'):
            xml_result={}
            object_name=object.find('name').text
            if object_name.split('_')[0]=='Text':
                content=object_name.split('_')[1]
                Xmin=int(object.find('bndbox').find('xmin').text)
                Ymin=int(object.find('bndbox').find('ymin').text)
                Xmax=int(object.find('bndbox').find('xmax').text)
                Ymax=int(object.find('bndbox').find('ymax').text)
                cv2.rectangle(im,(Xmin,Ymin),(Xmax,Ymax),color,2)
                font = cv2.FONT_HERSHEY_SIMPLEX  
                cv2.putText(im, object_name, (Xmin,Ymin - 7), font, 0.5, (0, 0, 250), 2)
                cv2.imshow('01',im)
                xml_result['id']=str(1200000+text_index)
                xml_result['content']=content
                xml_result['box']=str(Xmin)+","+str(Ymin)+";"+str(Xmax)+","+str(Ymax)
                self.text.append(xml_result)
                text_index=text_index+1
            elif object_name.split('_')[0]!='Text':
                type=object_name.split('_')[0]
                rotate=object_name.split('_')[1]
                Xmin=int(object.find('bndbox').find('xmin').text)
                Ymin=int(object.find('bndbox').find('ymin').text)
                Xmax=int(object.find('bndbox').find('xmax').text)
                Ymax=int(object.find('bndbox').find('ymax').text)
                cv2.rectangle(im,(Xmin,Ymin),(Xmax,Ymax),color,2)
                font = cv2.FONT_HERSHEY_SIMPLEX  
                cv2.putText(im, object_name, (Xmin,Ymin - 7), font, 0.5, (0, 0, 250), 2)
                cv2.imshow('01',im)
                xml_result['id']=str(1000000+label_index)
                xml_result['vol']=str(self.vol)
                xml_result['type']=type
                xml_result['box']=str(Xmin)+","+str(Ymin)+";"+str(Xmax)+","+str(Ymax)
                xml_result['rotate']=rotate
                xml_result['textId']=''
                self.label.append(xml_result)
                label_index=label_index+1
            else:
                continue
        cv2.imwrite('02.jpg', im)
    def new_xml_generate(self):
        '''
        该函数用于生成新的xml树
        '''
        self.old_xml_parse()
        result=ET.Element('result')
        result.text='\n'
        icons=ET.SubElement(result,'icons')
        icons.text='\n'
        icons.tail='\n'
        texts=ET.SubElement(result,'texts')
        texts.text='\n'
        for label in self.label:
            icon=ET.SubElement(icons,'icon')
            icon.attrib=label
            icon.tail='\n'
        for label in self.text:
            text=ET.SubElement(texts,'text')
            text.attrib=label
            text.tail='\n'
        self.indent(result,0)
        tree = ET.ElementTree(result)
        tree.write(self.save_path,encoding='utf-8')
    def indent(self,elem,level=0):
        '''
        该函数用于xml缩进

        Attributes:
             elem:要生成的xml树的根节点
             level:缩进水平，默认为两个空格
        '''
        i = "\n" + level*"\t"
        if len(elem):
            if not elem.text or not elem.text.strip():
                elem.text = i + "\t"
            if not elem.tail or not elem.tail.strip():
                elem.tail = i
            for elem in elem:
                self.indent(elem, level+1)
            if not elem.tail or not elem.tail.strip():
                elem.tail = i
        else:
            if level and (not elem.tail or not elem.tail.strip()):
                elem.tail = i

if __name__ == '__main__':
    path='old_xml'
    img_path='old_xml/6月12（标注更改版）/河南-鲁阳电厂主接线图_Copy1/河南-鲁阳电厂主接线图_Copy1.png'
    xml_path='old_xml/6月12（标注更改版）/河南-鲁阳电厂主接线图_Copy1/河南-鲁阳电厂主接线图_Copy1.xml'
    xml=Xml(xml_path,500,'new_xml/河南-鲁阳电厂主接线图_Copy1.xml',img_path)
    xml.old_xml_parse()

        for f in files:

                xml=Xml(os.path.join(root,f),500,'new_xml'+'/'+f)
                xml.new_xml_generate()
    '''
